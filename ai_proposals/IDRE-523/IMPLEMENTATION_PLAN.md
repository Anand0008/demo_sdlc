### `scripts/reset-2fa.ts` (create)
This script fulfills the operational request to reset 2FA for a specific team member by providing a reusable CLI tool that looks up a user by email and calls the existing `resetUser2FA` action.
```typescript
import { PrismaClient } from "@prisma/client";
import { resetUser2FA } from "@/lib/actions/admin-reset-2fa";

const prisma = new PrismaClient();

async function main() {
  const email = process.argv[2];

  if (!email) {
    console.error("Error: Please provide an email address.");
    console.error("Usage: npx tsx scripts/reset-2fa.ts <email>");
    process.exit(1);
  }

  console.log(`Looking up user with email: ${email}`);

  try {
    const user = await prisma.user.findUnique({
      where: { email },
    });

    if (!user) {
      console.error(`Error: User not found with email: ${email}`);
      process.exit(1);
    }

    console.log(`Found user: ${user.name || email} (ID: ${user.id})`);
    console.log("Attempting to reset 2FA...");

    // Call the existing admin action to reset 2FA
    const result = await resetUser2FA(user.id);

    if (result.success) {
      console.log(`Successfully reset 2FA for user ${email}.`);
    } else {
      console.error(`Failed to reset 2FA: ${result.error}`);
      process.exit(1);
    }
  } catch (error) {
    console.error("An unexpected error occurred:", error);
    process.exit(1);
  } finally {
    await prisma.$disconnect();
  }
}

main();
```
